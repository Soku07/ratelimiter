    %% ======================
    %% Orchestration
    %% ======================
    class RateLimitExecutor {
        -Map~String, RateLimitAlgorithm~ algorithms
        -Map~String, IdentityResolver~ resolvers
        -StorageProvider storage
        -PolicyRegistry policyRegistry
        +execute(HttpServletRequest req) RateLimitResult
    }

    %% ======================
    %% Identity & Multi-tenancy (The "Who")
    %% ======================
    class IdentityResolver {
        <<interface>>
        +resolve(HttpServletRequest req) String
    }
    
    class ClientUserCompositeResolver {
        -String clientIdHeader
        +resolve(HttpServletRequest req) String
    }
    
    class IpResolver {
        +resolve(HttpServletRequest req) String
    }

    IdentityResolver <|.. ClientUserCompositeResolver
    IdentityResolver <|.. IpResolver

    %% ======================
    %% Adaptive Rules (The "What")
    %% ======================
    class PolicyRegistry {
        -AtomicReference~RuleStore~ currentStore
        -RuleProvider ruleProvider
        +resolvePolicy(String path) Optional~AbstractRule~
        +refresh() void
    }

    class RuleProvider {
        <<interface>>
        +load() RuleStore
    }

    class YamlRuleProvider {
        -String yamlPath
        +load() RuleStore
    }

    RuleProvider <|.. YamlRuleProvider

    class RuleStore {
        <<interface>>
        +findBestMatch(String path) Optional~AbstractRule~
    }

    class OrderedListRuleStore {
        -List~AbstractRule~ sortedRules
        +findBestMatch(String path)
    }

    RuleStore <|.. OrderedListRuleStore

    %% ======================
    %% Models
    %% ======================
    class AbstractRule {
        <<abstract>>
        -String pathPattern
        -int priority
        -RateLimitPolicy policy
        +matches(String path)* boolean
    }

    class RateLimitPolicy {
        <<record>>
        -int limit
        -Duration window
        -String algorithmKey
        -String identityStrategy
    }

    %% ======================
    %% Engine & State (The "How" & "Where")
    %% ======================
    class RateLimitAlgorithm {
        <<interface>>
        +isAllowed(String key, RateLimitPolicy p, StorageProvider s) RateLimitResult
    }

    class StorageProvider {
        <<interface>>
        +incrementAndGet(String key, Duration ttl) long
        +setRaw(String key, Object val, Duration ttl) void
        +getRaw(String key) Object
    }

    %% ======================
    %% Relationships & Flow (Updated Connections)
    %% ======================
    RateLimitExecutor o-- IdentityResolver : selects
    RateLimitExecutor o-- RateLimitAlgorithm : selects
    RateLimitExecutor --> PolicyRegistry : queries
    
    %% RuleStore Wiring
    PolicyRegistry o-- RuleStore : manages (holds active instance)
    RuleProvider ..> RuleStore : load() produces
    
    OrderedListRuleStore o-- AbstractRule : contains
    AbstractRule *-- RateLimitPolicy : holds
    RateLimitAlgorithm ..> StorageProvider : state access
